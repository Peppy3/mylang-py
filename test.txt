# Vec3 is a value with type struct which is defined to a block of code
Vec3: struct {
    x: f32
    y: f32
    z: f32

    dot: (lhs: Vec3, rhs: Vec3) -> f32 {
        return lhs.x + rhs.x * lhs.y + rhs.y * lhs.z + rhs.y
    }
}
